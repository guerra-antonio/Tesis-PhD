\documentclass[tikz,border=12pt]{standalone}
\usepackage{amsmath}
\usepackage[T1]{fontenc}
\usepackage{tikz}
\usetikzlibrary{positioning,arrows.meta,backgrounds,fit,calc}

\definecolor{cblue}{RGB}{31,90,150}
\definecolor{cbluefill}{RGB}{225,236,247}
\definecolor{cgreen}{RGB}{40,120,55}
\definecolor{cgreenfill}{RGB}{222,240,220}
\definecolor{cpurple}{RGB}{95,50,135}
\definecolor{cpurplefill}{RGB}{236,228,245}
\definecolor{czerofill}{RGB}{248,246,251}
\definecolor{cpadfill}{RGB}{251,243,221}
\definecolor{cpadln}{RGB}{170,130,40}
\definecolor{cnavy}{RGB}{20,45,90}

\begin{document}
\begin{tikzpicture}[
  font=\small,
  >={Stealth[length=2.6mm,width=2.0mm]},
  cx/.style={minimum size=7mm, draw=cpurple!55, line width=0.4pt, fill=cpurplefill, anchor=center},
  cz/.style={minimum size=7mm, draw=cpurple!30, line width=0.4pt, fill=czerofill, anchor=center},
  cb/.style={minimum size=7mm, draw=cpadln!70, line width=0.4pt, fill=cpadfill, anchor=center},
  ck/.style={minimum size=7mm, draw=cgreen!75, line width=0.5pt, fill=cgreenfill, anchor=center},
  cy/.style={minimum size=7mm, draw=cblue!55, line width=0.4pt, fill=cbluefill, anchor=center},
]
\def\s{0.7}

% =================== INPUT 3x3 ===================
\foreach \r in {1,2,3}{\foreach \c in {1,2,3}{
  \node[cx] (I\r\c) at ({0+\c*\s},{4.4-\r*\s}) {\scriptsize$x_{\r\c}$};
}}
\draw[cpurple, line width=1.0pt] (0.35,4.05) rectangle (2.45,1.95);
\node[cnavy, font=\bfseries, align=center] at (1.4,4.85) {Input\\$3\times3$};

% step 1
\draw[cpurple, ->, line width=1.0pt] (2.6,3.0) -- (3.9,3.0);
\node[cpurple, align=center, font=\scriptsize, anchor=north] at (3.25,2.7) {\textbf{Step 1}\\insert \\zeros};

% =================== DILATED 5x5 ===================
\foreach \r in {1,...,5}{\foreach \c in {1,...,5}{
  \pgfmathtruncatemacro{\ro}{Mod(\r,2)}\pgfmathtruncatemacro{\co}{Mod(\c,2)}
  \pgfmathtruncatemacro{\xi}{(\r+1)/2}\pgfmathtruncatemacro{\xj}{(\c+1)/2}
  \ifnum\ro=1 \ifnum\co=1
     \node[cx] at ({3.7+\c*\s},{5.1-\r*\s}) {\scriptsize$x_{\xi\xj}$};
  \else \node[cz] at ({3.7+\c*\s},{5.1-\r*\s}) {\scriptsize$0$};\fi
  \else \node[cz] at ({3.7+\c*\s},{5.1-\r*\s}) {\scriptsize$0$};\fi
}}
\draw[cpurple, line width=1.0pt] (4.05,4.75) rectangle (7.55,1.25);
\node[cnavy, font=\bfseries, align=center, anchor=north] at (5.8,1.05) {Dilated input\\$5\times5$};

% step 2
\draw[cpurple, ->, line width=1.0pt] (7.7,3.0) -- (9.0,3.0);
\node[cpadln!90!black, align=center, font=\scriptsize, anchor=north] at (8.35,2.7) {\textbf{Step 2}\\zero\\ padding};

% =================== ZERO-PADDED 7x7 ===================
\foreach \r in {1,...,7}{\foreach \c in {1,...,7}{
  \pgfmathtruncatemacro{\bd}{((\r==1)||(\r==7)||(\c==1)||(\c==7)) ? 1 : 0}
  \pgfmathtruncatemacro{\re}{Mod(\r,2)}\pgfmathtruncatemacro{\ce}{Mod(\c,2)}
  \pgfmathtruncatemacro{\xi}{\r/2}\pgfmathtruncatemacro{\xj}{\c/2}
  \ifnum\bd=1
     \node[cb] (P\r\c) at ({8.8+\c*\s},{5.8-\r*\s}) {\scriptsize$0$};
  \else
     \ifnum\re=0 \ifnum\ce=0
        \node[cx] (P\r\c) at ({8.8+\c*\s},{5.8-\r*\s}) {\scriptsize$x_{\xi\xj}$};
     \else \node[cz] (P\r\c) at ({8.8+\c*\s},{5.8-\r*\s}) {\scriptsize$0$};\fi
     \else \node[cz] (P\r\c) at ({8.8+\c*\s},{5.8-\r*\s}) {\scriptsize$0$};\fi
  \fi
}}
\draw[cblue!1, line width=0.1pt] (9.15,5.45) rectangle (14.05,0.55);
% emphasize the padding ring
\draw[cpadln, line width=1.0pt] (9.15,5.45) rectangle (14.05,0.55);
\draw[cpadln, dashed, line width=0.8pt] (9.85,4.75) rectangle (13.35,1.25);
\node[cnavy, font=\bfseries, align=center, anchor=north] at (11.6,0.35) {Zero-padded input\\$7\times7$};
\node[cpadln!90!black, font=\itshape\scriptsize, anchor=west] at (14.2,5.3) {added border};
\draw[cpadln, line width=0.5pt] (14.18,5.3) -- (13.7,5.1);
% receptive-field patch (top-left 3x3) -- border only
\draw[cgreen, line width=1.4pt] (9.15,5.45) rectangle (11.25,3.35);

% =================== KERNEL (flat, separate) above the patch ===================
\foreach \r in {1,2,3}{\foreach \c in {1,2,3}{
  \node[ck] (K\r\c) at ({10.1+\c*\s},{8.9-\r*\s}) {\scriptsize$k_{\r\c}$};
}}
\draw[cgreen, line width=1.1pt] (10.45,8.55) rectangle (12.55,6.45);
\node[cgreen, font=\bfseries, align=center] at (11.5,9.0) {Kernel ($K$), $3\times3$};
\foreach \pp/\kk in {{9.15,5.45}/{10.45,8.55}, {11.25,5.45}/{12.55,8.55},
                     {11.25,3.35}/{12.55,6.45}, {9.15,3.35}/{10.45,6.45}}{
  \draw[cgreen!75, dashed, line width=0.6pt] (\pp) -- (\kk);
}

% step 3
\draw[cgreen, ->, line width=1.0pt] (14.2,3.0) -- (15.5,3.0);
\node[cgreen, align=center, font=\scriptsize, anchor=south] at (14.85,3.35) {\textbf{Step 3}\\convolution};

% =================== OUTPUT 5x5 ===================
\fill[cblue!20] (15.65,4.75) rectangle (16.35,4.05);
\foreach \r in {1,...,5}{\foreach \c in {1,...,5}{
  \pgfmathtruncatemacro{\isone}{((\r==1)&&(\c==1)) ? 1 : 0}
  \ifnum\isone=1
    \node[cy, fill=none, draw=cblue, line width=1.4pt] (O\r\c) at ({15.3+\c*\s},{5.1-\r*\s}) {\scriptsize$y_{\r\c}$};
  \else
    \node[cy] (O\r\c) at ({15.3+\c*\s},{5.1-\r*\s}) {\scriptsize$y_{\r\c}$};
  \fi
}}
\draw[cblue, line width=1.0pt] (15.65,4.75) rectangle (19.15,1.25);
\node[cnavy, font=\bfseries, align=center] at (17.4,5.3) {Output\\$5\times5$};

\end{tikzpicture}
\end{document}
